$desktopRoot = "C:\Users\Public\Desktop"
$profileName = "FreeFire"

function Resolve-MacroRoot {
    param(
        [string]$BaseDir,
        [string]$ProfileName
    )

    $candidates = New-Object System.Collections.Generic.List[string]

    $rootScript = Join-Path $BaseDir "smart_macro.py"
    if (Test-Path $rootScript) {
        $candidates.Add($BaseDir)
    }

    Get-ChildItem -Path $BaseDir -Filter "smart_macro.py" -Recurse -File -ErrorAction SilentlyContinue |
        ForEach-Object {
            $candidates.Add($_.Directory.FullName)
        }

    foreach ($candidate in ($candidates | Select-Object -Unique)) {
        $profileConfig = Join-Path $candidate "profiles\$ProfileName\config.json"
        if (Test-Path $profileConfig) {
            return $candidate
        }
    }

    foreach ($candidate in ($candidates | Select-Object -Unique)) {
        if (Test-Path (Join-Path $candidate "smart_macro.py")) {
            return $candidate
        }
    }

    return $null
}

function Resolve-ScreenShareRoot {
    param([string]$BaseDir)

    $scriptMatches = Get-ChildItem -Path $BaseDir -Filter "Share your screen on Telegram.py" -Recurse -File -ErrorAction SilentlyContinue
    if ($scriptMatches) {
        return ($scriptMatches | Select-Object -First 1).Directory.FullName
    }

    return $null
}

function Get-SessionObjects {
    param([string[]]$Lines)

    $sessions = @()
    foreach ($line in $Lines) {
        if ($line -match '^\s*>?\s*(?<user>\S+)\s+(?<session>\S+)\s+(?<id>\d+)\s+(?<state>Active|Disc)\b') {
            $sessions += [pscustomobject]@{
                User        = $matches['user']
                SessionName = $matches['session']
                Id          = [int]$matches['id']
                State       = $matches['state']
            }
        }
    }

    return $sessions
}

$macroRoot = Resolve-MacroRoot -BaseDir $desktopRoot -ProfileName $profileName
if (-not $macroRoot) {
    Write-Error "Could not locate smart_macro.py under $desktopRoot"
    exit 1
}

$macroScript = Join-Path $macroRoot "smart_macro.py"
$profileConfig = Join-Path $macroRoot "profiles\$profileName\config.json"
$envFilePath = Join-Path $macroRoot ".env"
$macroLogPath = Join-Path $macroRoot "smart_macro_runtime.log"
$launcherPath = Join-Path $macroRoot "launch_smart_macro.cmd"
$screenShareEnabled = -not [string]::IsNullOrEmpty($env:TELEGRAM_TOKEN) -and -not [string]::IsNullOrEmpty($env:TELEGRAM_CHAT_ID)
$screenShareRoot = $null
$screenShareScript = $null
$screenShareEnvPath = $null
$screenShareLogPath = $null
$screenShareLauncherPath = $null

Write-Host "Macro root resolved to: $macroRoot"
Write-Host "Macro script path: $macroScript"
Write-Host "Profile config path: $profileConfig"
Write-Host "Runtime log path: $macroLogPath"

if (-not (Test-Path $profileConfig)) {
    Write-Error "Profile '$profileName' was not found at $profileConfig"
    Write-Host "Desktop folders detected:"
    Get-ChildItem -Path $desktopRoot -Directory -ErrorAction SilentlyContinue | ForEach-Object { Write-Host " - $($_.FullName)" }
    exit 1
}

Write-Host "Creating temporary .env file in Macro directory..."
"TARGET_EMAIL=$env:TARGET_EMAIL" | Out-File -FilePath $envFilePath -Encoding utf8

$pythonPath = (Get-Command python -ErrorAction SilentlyContinue).Source
if (-not $pythonPath) {
    $pythonPath = "python"
}
Write-Host "Python path: $pythonPath"

if (-not (Test-Path "C:\Users\Public\PsExec64.exe")) {
    Write-Error "PsExec64.exe was not found at C:\Users\Public\PsExec64.exe"
    exit 1
}

Write-Host "Current sessions:"
$rawSessions = query user
$rawSessions | ForEach-Object { Write-Host $_ }

$sessions = Get-SessionObjects -Lines $rawSessions
if (-not $sessions) {
    Write-Error "Failed to parse any user sessions from 'query user'."
    exit 1
}

$preferredSession = $sessions | Where-Object { $_.User -eq "TOOLBOXLAP" -and $_.State -eq "Active" } | Select-Object -First 1
if (-not $preferredSession) {
    $preferredSession = $sessions | Where-Object { $_.State -eq "Active" } | Select-Object -First 1
}

if (-not $preferredSession) {
    Write-Error "No ACTIVE session was found. The macro needs an interactive desktop."
    exit 1
}

$pinnedSessionId = $null
if ($env:TOOLBOXLAP_SESSION_ID) {
    $pinnedSessionId = [int]$env:TOOLBOXLAP_SESSION_ID
    Write-Host "Workflow pinned session ID detected: $pinnedSessionId"

    $exactSession = $sessions | Where-Object { $_.Id -eq $pinnedSessionId } | Select-Object -First 1
    if ($exactSession -and $exactSession.State -eq "Active") {
        $preferredSession = $exactSession
        Write-Host "Using pinned GUI session from workflow."
    } else {
        Write-Host "Pinned session is missing or not active anymore. Falling back to live detection..."
    }
}

Write-Host "Selected session: User=$($preferredSession.User) Session=$($preferredSession.SessionName) ID=$($preferredSession.Id) State=$($preferredSession.State)"

if ($screenShareEnabled) {
    $screenShareRoot = Resolve-ScreenShareRoot -BaseDir $desktopRoot
    if (-not $screenShareRoot) {
        Write-Error "Telegram screen-share bundle was enabled but 'Share your screen on Telegram.py' was not found under $desktopRoot"
        exit 1
    }

    $screenShareScript = Join-Path $screenShareRoot "Share your screen on Telegram.py"
    $screenShareEnvPath = Join-Path $screenShareRoot ".env"
    $screenShareLogPath = Join-Path $screenShareRoot "telegram_screen_share.log"
    $screenShareLauncherPath = Join-Path $screenShareRoot "launch_telegram_screen_share.cmd"

    Write-Host "Screen-share root resolved to: $screenShareRoot"
    Write-Host "Screen-share script path: $screenShareScript"
    Write-Host "Screen-share log path: $screenShareLogPath"

    $screenShareEnvLines = @(
        "TELEGRAM_TOKEN=$env:TELEGRAM_TOKEN",
        "TELEGRAM_CHAT_ID=$env:TELEGRAM_CHAT_ID"
    )
    if (-not [string]::IsNullOrEmpty($env:SCREEN_SHARE_NOTICE_TEXT)) {
        $screenShareEnvLines += "SCREEN_SHARE_NOTICE_TEXT=$env:SCREEN_SHARE_NOTICE_TEXT"
    }
    Set-Content -Path $screenShareEnvPath -Value $screenShareEnvLines -Encoding utf8

    $screenShareLauncherContent = @"
@echo off
cd /d "$screenShareRoot"
set "SCREEN_SHARE_SKIP_ADMIN=1"
echo [LAUNCH] %date% %time% >> "$screenShareLogPath"
"$pythonPath" "$screenShareScript" >> "$screenShareLogPath" 2>&1
echo [EXITCODE] %errorlevel% >> "$screenShareLogPath"
"@
    Set-Content -Path $screenShareLauncherPath -Value $screenShareLauncherContent -Encoding ascii

    Add-Content -Path $screenShareLogPath -Value "============================================================"
    Add-Content -Path $screenShareLogPath -Value "PowerShell launcher started at $(Get-Date -Format s)"
    Add-Content -Path $screenShareLogPath -Value "Pinned session ID: $($preferredSession.Id)"

    Write-Host "Launching Telegram screen-share bot in the same interactive desktop..."
    C:\Users\Public\PsExec64.exe -s -i $preferredSession.Id -d -w "$screenShareRoot" -accepteula cmd.exe /c `"$screenShareLauncherPath`"
}

$launcherContent = @"
@echo off
cd /d "$macroRoot"
echo [LAUNCH] %date% %time% >> "$macroLogPath"
"$pythonPath" "$macroScript" --auto-run "$profileName" --email "$env:TARGET_EMAIL" >> "$macroLogPath" 2>&1
echo [EXITCODE] %errorlevel% >> "$macroLogPath"
"@
Set-Content -Path $launcherPath -Value $launcherContent -Encoding ascii

Add-Content -Path $macroLogPath -Value "============================================================"
Add-Content -Path $macroLogPath -Value "PowerShell launcher started at $(Get-Date -Format s)"
Add-Content -Path $macroLogPath -Value "Macro root: $macroRoot"
Add-Content -Path $macroLogPath -Value "Profile config: $profileConfig"
Add-Content -Path $macroLogPath -Value "Selected session ID: $($preferredSession.Id)"

Write-Host "Launching Smart Macro in the interactive desktop..."
C:\Users\Public\PsExec64.exe -s -i $preferredSession.Id -d -w "$macroRoot" -accepteula cmd.exe /c `"$launcherPath`"

Start-Sleep -Seconds 10

if ($screenShareEnabled) {
    $screenShareProcesses = Get-CimInstance Win32_Process -ErrorAction SilentlyContinue | Where-Object {
        $_.CommandLine -and $_.CommandLine -like "*Share your screen on Telegram.py*"
    }

    if ($screenShareProcesses) {
        Write-Host "Telegram screen-share bot is running. Matching processes:"
        $screenShareProcesses | ForEach-Object {
            Write-Host " - PID=$($_.ProcessId) Session=$($_.SessionId) CommandLine=$($_.CommandLine)"
        }
    } else {
        Write-Error "Telegram screen-share bot exited quickly or did not launch."
        if (Test-Path $screenShareLogPath) {
            Write-Host "Last lines from ${screenShareLogPath}:"
            Get-Content $screenShareLogPath -Tail 80 | ForEach-Object { Write-Host $_ }
        }
        exit 1
    }
}

$macroProcesses = Get-CimInstance Win32_Process -ErrorAction SilentlyContinue | Where-Object {
    $_.CommandLine -and $_.CommandLine -like "*smart_macro.py*"
}

if ($macroProcesses) {
    Write-Host "Smart Macro is running. Matching processes:"
    $macroProcesses | ForEach-Object {
        Write-Host " - PID=$($_.ProcessId) Session=$($_.SessionId) CommandLine=$($_.CommandLine)"
    }
} else {
    Write-Error "Smart Macro exited quickly or did not launch."
    if (Test-Path $macroLogPath) {
        Write-Host "Last lines from ${macroLogPath}:"
        Get-Content $macroLogPath -Tail 80 | ForEach-Object { Write-Host $_ }
    }
    exit 1
}
