# ==========================================
# Google Drive Download Helper Function
# ==========================================
function Download-FromGoogleDrive {
    param(
        [string]$Url,
        [string]$OutFile
    )
    
    # Extract file ID from various Google Drive URL formats
    $fileId = $null
    if ($Url -match "id=([a-zA-Z0-9_-]+)") {
        $fileId = $matches[1]
    } elseif ($Url -match "/d/([a-zA-Z0-9_-]+)") {
        $fileId = $matches[1]
    }
    
    if (-not $fileId) {
        Write-Host "Not a Google Drive URL, downloading directly..."
        Invoke-WebRequest -Uri $Url -OutFile $OutFile
        return
    }
    
    Write-Host "Detected Google Drive file ID: $fileId"
    $downloadUrl = "https://drive.google.com/uc?export=download&id=$fileId"
    
    # First request - may get a virus scan confirmation page
    $session = New-Object Microsoft.PowerShell.Commands.WebRequestSession
    $response = Invoke-WebRequest -Uri $downloadUrl -WebSession $session -UseBasicParsing -MaximumRedirection 5
    
    # Check if we got a confirmation page (large file virus scan warning)
    $confirmMatch = [regex]::Match($response.Content, 'confirm=([0-9a-zA-Z_-]+)')
    if ($confirmMatch.Success) {
        $confirmCode = $confirmMatch.Groups[1].Value
        Write-Host "Large file detected, confirming download..."
        $downloadUrl = "https://drive.google.com/uc?export=download&confirm=$confirmCode&id=$fileId"
        Invoke-WebRequest -Uri $downloadUrl -WebSession $session -OutFile $OutFile -UseBasicParsing
    } else {
        # Check for uuid confirm token (newer Google Drive format)
        $uuidMatch = [regex]::Match($response.Content, 'uuid=([0-9a-zA-Z_-]+)')
        if ($uuidMatch.Success) {
            $uuid = $uuidMatch.Groups[1].Value
            Write-Host "Large file detected (uuid), confirming download..."
            $downloadUrl = "https://drive.google.com/uc?export=download&confirm=t&uuid=$uuid&id=$fileId"
            Invoke-WebRequest -Uri $downloadUrl -WebSession $session -OutFile $OutFile -UseBasicParsing
        } elseif ($response.Content.Length -gt 100000 -or $response.Headers['Content-Type'] -notlike '*text/html*') {
            # Small file - response IS the file content
            [System.IO.File]::WriteAllBytes($OutFile, $response.Content)
        } else {
            # Fallback: try direct download with confirm=t
            Write-Host "Trying direct confirmed download..."
            $downloadUrl = "https://drive.google.com/uc?export=download&confirm=t&id=$fileId"
            Invoke-WebRequest -Uri $downloadUrl -OutFile $OutFile -UseBasicParsing
        }
    }
    
    $fileSize = (Get-Item $OutFile -ErrorAction SilentlyContinue).Length
    Write-Host "Downloaded: $OutFile ($([math]::Round($fileSize/1MB, 2)) MB)"
}
