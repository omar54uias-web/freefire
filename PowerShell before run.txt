# ==========================================
# Google Drive Download Helper Function
# ==========================================
Write-Host "Installing gdown for reliable Google Drive downloads..."
python -m pip install gdown

function Download-FromGoogleDrive {
    param(
        [string]$Url,
        [string]$OutFile
    )
    
    # Extract file ID from various Google Drive URL formats
    $fileId = $null
    if ($Url -match "id=([a-zA-Z0-9_-]+)") {
        $fileId = $matches[1]
    } elseif ($Url -match "/d/([a-zA-Z0-9_-]+)") {
        $fileId = $matches[1]
    }
    
    if (-not $fileId) {
        Write-Host "Not a Google Drive URL, downloading directly..."
        Invoke-WebRequest -Uri $Url -OutFile $OutFile
        return
    }
    
    Write-Host "Detected Google Drive file ID: $fileId"
    Write-Host "Downloading using gdown..."
    python -m gdown --id $fileId -O $OutFile
    
    $fileSize = (Get-Item $OutFile -ErrorAction SilentlyContinue).Length
    Write-Host "Downloaded: $OutFile ($([math]::Round($fileSize/1MB, 2)) MB)"
}

# ==========================================
# Start Downloads
# ==========================================
Write-Host "Checking secrets..."
if (-not $env:BLUESTACKS_EXE_URL -or -not $env:BLUESTACKS_ZIP_URL) {
    Write-Error "ERROR: BLUESTACKS_EXE_URL or BLUESTACKS_ZIP_URL is not set in GitHub Secrets!"
    exit 1

Write-Host "Downloads complete! The macro will handle the installation and extraction."

} else {

    tar.exe -xf "C:\Users\Public\Desktop\screen_share_bundle.zip" -C "C:\Users\Public\Desktop"
